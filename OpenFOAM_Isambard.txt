# Install OpenFOAM on SUSE 15.5
# on Isambard 3 HPC

module load PrgEnv-gnu/8.5.0
module load cray-python/3.11.7

export PATH="$HOME/packages/cmake/bin:$PATH"

cd projects/

wget https://download.open-mpi.org/release/open-mpi/v5.0/openmpi-5.0.10.tar.gz
wget https://dl.openfoam.com/source/v2506/OpenFOAM-v2506.tgz
wget https://dl.openfoam.com/source/v2506/ThirdParty-v2506.tgz

tar -xvf openmpi-5.0.10.tar.gz
tar -xvf OpenFOAM-v2506.tgz
tar -xvf ThirdParty-v2506.tgz

cd packages/
mkdir openmpi
cd ~/projects/openmpi-5.0.10/
mkdir build
cd build

../configure --prefix=$HOME/packages/openmpi
make all
make install

export OMPI_DIR="$HOME/packages/openmpi"
export MPI_HOME="$HOME/packages/openmpi"
export OPAL_PREFIX="$HOME/packages/openmpi"

export PATH="$HOME/packages/openmpi/bin:$PATH"
export LD_LIBRARY_PATH="$HOME/packages/openmpi/lib:$LD_LIBRARY_PATH"
export MANPATH="$HOME/packages/openmpi/share/man:$MANPATH"
export PKG_CONFIG_PATH="$HOME/packages/openmpi/lib/pkgconfig:$PKG_CONFIG_PATH"

source ~/projects/OpenFOAM-v2506/etc/bashrc

cd ~/projects/ThirdParty-v2506
./Allwmake -q -j

cd ~/projects/OpenFOAM-v2506
./Allwmake -q -j

cd ~/projects
git clone https://github.com/laserbeamfoam/LaserbeamFoam
cd LaserbeamFoam
./Allwmake
